*** mySQL 连接
***** mySQL 服务器启动命令
#+BEGIN_EXAMPLE
  $ mysql.server start
  $ mysql.server stop
  $ mysql.server restart
#+END_EXAMPLE
***** mySQL 连接服务器 
#+BEGIN_EXAMPLE
  $ mysql -u root -p
if have password , then enter....
  $ yourpassword

#+END_EXAMPLE

#+BEGIN_SRC sql

mysql> quit;

$ mysql -u tom -p
$ yourpassword
